import boto3

s3_client = boto3.client('s3')

def lambda_handler(event, context):
    bucket_name = 'chipi'
    fileName = event["fileName"]
    data = s3_client.get_object(Bucket = bucket_name, Key = fileName)["Body"].read().decode("utf-8")
    
    data = [str(eval(line)) for line in data.split("\n")]
    print(data)
    new_file_key = fileName.replace('.txt', '_updated.txt')
        
    s3_client.put_object(Key=new_file_key, Bucket="chaapa", Body="\n".join(data))
