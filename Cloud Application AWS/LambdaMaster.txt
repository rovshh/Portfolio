import json
import boto3
awsLambda = boto3.client("lambda")
awsClient = boto3.client("s3")
def lambda_handler(event, context):
    fileObj = event["Records"][0]
    fileName = str(fileObj["s3"]["object"]["key"])
    contextData = json.dumps({"fileName": fileName})
    if fileName.startswith('A'):
        awsLambda.invoke(
            FunctionName = "arn:aws:lambda:eu-central-1:975844241338:function:LambdaA",
            InvocationType = "Event",
            Payload = contextData
        )
        
    elif fileName.startswith('B'):
        awsLambda.invoke(
            FunctionName = "arn:aws:lambda:eu-central-1:975844241338:function:LambdaB",
            InvocationType = "Event",
            Payload = contextData
        )
        
    elif fileName.startswith('C'):
        awsLambda.invoke(
            FunctionName = "arn:aws:lambda:eu-central-1:975844241338:function:LambdaC",
            InvocationType = "Event",
            Payload = contextData
        )
